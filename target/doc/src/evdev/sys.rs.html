128/// ioctl: "get abs value/limits"
129///
130/// `abs` should be one of the "Absolute axes" values defined in the Linux kernel headers.
131/// In modern (5.11) kernels these are in `include/uapi/linux/input-event-codes.h`, and in older
132/// kernels these defines can be found in `include/uapi/linux/input.h`
133///
134/// # Panics
135///
136/// Calling this with a value greater than the kernel-defined `ABS_MAX` (typically 0x3f) will panic.
137///
138/// # Safety
139///
140/// 'abs' must be a valid axis number and supported by the device, otherwise the behavior is
141/// undefined.
142pub unsafe fn eviocgabs(
143    fd: ::libc::c_int,
144    abs: u32,
145    buf: &mut input_absinfo,
146) -> ::nix::Result<::libc::c_int> {
147    assert!(abs <= 0x3f);
148    convert_ioctl_res!(::nix::libc::ioctl(
149        fd,
150        request_code_read!(b'E', 0x40 + abs, ::std::mem::size_of::<input_absinfo>()),
151        buf as *mut input_absinfo
152    ))
153}
